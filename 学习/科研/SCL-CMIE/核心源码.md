columsNames = [

    'land', 'wrong_fragment', 'urgent', 'hot', 'num_failed_logins', 'logged_in',

attack_mapping = {

    'smurf': 'DOS', 'teardrop': 'DOS', 'apache2': 'DOS', 'udpstorm': 'DOS',

    # PROBE攻击  
    'satan': 'PROBE', 'ipsweep': 'PROBE', 'nmap': 'PROBE', 'portsweep': 'PROBE',

    # R2L攻击

    # U2R攻击

# 将class列映射到5分类  
dataTrain['class'] = dataTrain['class'].map(attack_mapping)

categorical_columns = ['protocol_type', 'service', 'flag']  
  
# 先对离散特征进行factorize编码（在划分x_train和x_test之前）  
for col in categorical_columns:  
    dataTrain[col], unique_categories = pd.factorize(dataTrain[col]) # 对训练集进行factorize  
    category_mapping = {cat: idx for idx, cat in enumerate(unique_categories)} # 创建映射字典

# 现在再划分特征和目标变量  
x_train = dataTrain.drop('class', axis=1)  
y_train = dataTrain['class']

# 对数值特征进行标准化（对每一列特征进行）  
numerical_columns = [col for col in x_train.columns if col not in categorical_columns]  
scaler = StandardScaler()  
  
x_train[numerical_columns] = scaler.fit_transform(x_train[numerical_columns])  
x_test[numerical_columns] = scaler.transform(x_test[numerical_columns])  
  
# 创建指示函数  
y_train_binary = np.where(y_train == 'NORMAL', 0, 1)

# 2. 计算各类别的平均特征向量（使用训练集计算）  
class_labels = ['NORMAL', 'DOS', 'PROBE', 'U2R', 'R2L']  
class_mean_vectors = {}  
  
for class_label in class_labels:  
    # 获取当前类别的所有样本  
    class_mask = (y_train == class_label)  
    if np.sum(class_mask) > 0:  
        # 计算该类别所有样本的特征平均值  
        class_data = x_train[class_mask]  
        mean_vector = class_data.mean(axis=0).values  # 转换为numpy数组  
        class_mean_vectors[class_label] = mean_vector  
        print(f"类别 {class_label} 的平均特征向量形状: {mean_vector.shape}, 样本数量: {np.sum(class_mask)}")  
    else:  
        print(f"警告: 训练集中没有找到类别 {class_label} 的样本")  
        class_mean_vectors[class_label] = np.zeros(x_train.shape[1])  
  
def create_label_vectors(labels, mean_vectors_dict):  
    """为标签列表创建对应的平均特征向量表示"""  
    label_vectors = []  
    for label in labels:  
        label_vectors.append(mean_vectors_dict[label])  
    return np.array(label_vectors)  
  
# 为训练集和测试集创建标签向量  
y_train_vectors = create_label_vectors(y_train, class_mean_vectors)  
y_test_vectors = create_label_vectors(y_test, class_mean_vectors)  
  
print("\n各类别平均特征向量示例 (前5个特征):")  
for class_label, mean_vector in class_mean_vectors.items():  
    print(f"{class_label}: {mean_vector[:5]}")  
  
# 转换为PyTorch张量  
x_train_tensor = torch.FloatTensor(x_train.values)  
y_train_vectors_tensor = torch.FloatTensor(y_train_vectors)  
y_train_binary_tensor = torch.FloatTensor(y_train_binary)  
x_test_tensor = torch.FloatTensor(x_test.values)  
y_test_vectors_tensor = torch.FloatTensor(y_test_vectors)

# 创建数据加载器  
batch_size = 256  # 可以根据你的硬件调整批次大小  
train_dataset = TensorDataset(x_train_tensor, y_train_vectors_tensor, y_train_binary_tensor)  
train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)  
  
test_dataset = TensorDataset(x_test_tensor, y_test_vectors_tensor, y_test_binary_tensor)  
test_loader = DataLoader(test_dataset, batch_size=batch_size, shuffle=False)  
  
# 保存预处理数据  
filename_save = 'NSL_KDD_Load.pkl'  
dill.dump_session(filename_save)

# Bidirectional Anchor Contrastive Loss  
class BACLoss(nn.Module):  
    def __init__(self, device, temperature=0.1):  
        super(BACLoss, self).__init__()  
        self.device = device  
        self.temperature = temperature  
      
    def forward(self, logits, labels):  
        logits /= self.temperature  
        logits_normalX = logits[(labels == 0).squeeze()]  
        logits_normalX_normalY = logits_normalX[:, (labels == 0).squeeze()]  
        logits_normalX_abnormalY = logits_normalX[:, (labels == 1).squeeze()]  
        denominatorX = torch.exp(logits_normalX_normalY) + torch.sum(torch.exp(logits_normalX_abnormalY), axis=1, keepdims=True)  
        log_probs_X = logits_normalX_normalY - torch.log(denominatorX)  
        loss_X = -log_probs_X  
        loss_X = loss_X.mean()  
          
        logits_normalY = logits[:, (labels == 0).squeeze()]  
        logits_normalY_normalX = logits_normalY[(labels == 0).squeeze()]  
        logits_normalY_abnormalX = logits_normalY[(labels == 1).squeeze()]  
        denominatorY = torch.exp(logits_normalY_normalX) + torch.sum(torch.exp(logits_normalY_abnormalX), axis=0, keepdims=True)  
        log_probs_Y = logits_normalY_normalX - torch.log(denominatorY)  
        loss_Y = -log_probs_Y  
        loss_Y = loss_Y.mean()  
          
        loss = loss_X + loss_Y  
        return loss  
  
class Encoder(nn.Module):  
    def __init__(self, input_dim):  
        super(Encoder, self).__init__()  
        nearest_power_of_2 = 2 ** round(math.log2(input_dim))  
        second_layer_size = nearest_power_of_2 // 2  
        third_layer_size = nearest_power_of_2 // 4  
        self.network = nn.Sequential(  
            nn.Linear(input_dim, second_layer_size),  
            nn.BatchNorm1d(second_layer_size),  
            nn.ReLU(),  
            nn.Dropout(0.2),  
            nn.Linear(second_layer_size, third_layer_size),  
            nn.BatchNorm1d(third_layer_size),  
            nn.ReLU(),  
            nn.Linear(third_layer_size, 32)  
        )  
      
    def forward(self, x):  
        return self.network(x)  
  
class ContrastiveModel(nn.Module):  
    def __init__(self, feature_dim, label_dim):  
        super(ContrastiveModel, self).__init__()  
        self.feature_encoder = Encoder(feature_dim)  
        self.label_encoder = Encoder(label_dim)  
      
    def forward(self, features, labels):  
        z_x = self.feature_encoder(features)     
        z_y = self.label_encoder(labels)        
        z_x = F.normalize(z_x, p=2, dim=1)  
        z_y = F.normalize(z_y, p=2, dim=1)  
        similarity_matrix = torch.matmul(z_x, z_y.T)   
        return similarity_matrix  
  
# 获取特征和标签的维度  
feature_dim = x_train.shape[1]  
label_dim = y_train_vectors.shape[1]  
  
# 模型初始化  
model = ContrastiveModel(feature_dim, label_dim).to(device)  
criterion = BACLoss(device=device, temperature=0.1)  
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)  
  
# 分批次训练函数  
def train_model(model, train_loader, criterion, optimizer, device, epochs=50):  
    model.train()  
    for epoch in range(epochs):  
        total_loss = 0.0  
        batch_count = 0  
        for batch_idx, (features, label_vectors, binary_labels) in enumerate(train_loader):  
            features = features.to(device)  
            label_vectors = label_vectors.to(device)  
            binary_labels = binary_labels.to(device)  
            optimizer.zero_grad()  
            similarity_matrix = model(features, label_vectors)  
            loss = criterion(similarity_matrix, binary_labels)  
            loss.backward()  
            optimizer.step()  
            total_loss += loss.item()  
            batch_count += 1  
        avg_loss = total_loss / batch_count  
        print(f'Epoch {epoch+1}/{epochs} completed. Average Loss: {avg_loss:.6f}')  
      
    return model  
  
# 评估函数  
def evaluate_model(model, test_loader, device):  
    model.eval()  
    total_loss = 0.0  
    batch_count = 0  
    with torch.no_grad():  
        for features, label_vectors, binary_labels in test_loader:  
            features = features.to(device)  
            label_vectors = label_vectors.to(device)  
            binary_labels = binary_labels.to(device)  
            similarity_matrix = model(features, label_vectors)  
            loss = criterion(similarity_matrix, binary_labels)  
            total_loss += loss.item()  
            batch_count += 1  
    avg_loss = total_loss / batch_count  
    print(f'Test Loss: {avg_loss:.6f}')  
    return avg_loss  
  
# 开始训练  
print("开始分批次训练...")  
trained_model = train_model(model, train_loader, criterion, optimizer, device, epochs=50)  
  
# 评估模型  
print("开始评估模型...")  
test_loss = evaluate_model(trained_model, test_loader, device)
