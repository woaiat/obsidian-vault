categorical_columns = ['protocol_type', 'service', 'flag']  
  
# 先对离散特征进行factorize编码（在划分x_train和x_test之前）  
for col in categorical_columns:

# Bidirectional Anchor Contrastive Loss  
class BACLoss(nn.Module):  
    def __init__(self, device, temperature=0.1):  
        super(BACLoss, self).__init__()  
        self.device = device  
        self.temperature = temperature  
      
    def forward(self, logits, labels):  
        logits /= self.temperature  
        logits_normalX = logits[(labels == 0).squeeze()]  
        logits_normalX_normalY = logits_normalX[:, (labels == 0).squeeze()]  
        logits_normalX_abnormalY = logits_normalX[:, (labels == 1).squeeze()]  
        denominatorX = torch.exp(logits_normalX_normalY) + torch.sum(torch.exp(logits_normalX_abnormalY), axis=1, keepdims=True)  
        log_probs_X = logits_normalX_normalY - torch.log(denominatorX)  
        loss_X = -log_probs_X  
        loss_X = loss_X.mean()  
          
        logits_normalY = logits[:, (labels == 0).squeeze()]  
        logits_normalY_normalX = logits_normalY[(labels == 0).squeeze()]  
        logits_normalY_abnormalX = logits_normalY[(labels == 1).squeeze()]  
        denominatorY = torch.exp(logits_normalY_normalX) + torch.sum(torch.exp(logits_normalY_abnormalX), axis=0, keepdims=True)  
        log_probs_Y = logits_normalY_normalX - torch.log(denominatorY)  
        loss_Y = -log_probs_Y  
        loss_Y = loss_Y.mean()  
          
        loss = loss_X + loss_Y  
        return loss  
  
class Encoder(nn.Module):  
    def __init__(self, input_dim):  
        super(Encoder, self).__init__()  
        nearest_power_of_2 = 2 ** round(math.log2(input_dim))  
        second_layer_size = nearest_power_of_2 // 2  
        third_layer_size = nearest_power_of_2 // 4  
        self.network = nn.Sequential(  
            nn.Linear(input_dim, second_layer_size),  
            nn.BatchNorm1d(second_layer_size),  
            nn.ReLU(),  
            nn.Dropout(0.2),  
            nn.Linear(second_layer_size, third_layer_size),  
            nn.BatchNorm1d(third_layer_size),  
            nn.ReLU(),  
            nn.Linear(third_layer_size, 32)  
        )  
      
    def forward(self, x):  
        return self.network(x)  
  
class ContrastiveModel(nn.Module):  
    def __init__(self, feature_dim, label_dim):  
        super(ContrastiveModel, self).__init__()  
        self.feature_encoder = Encoder(feature_dim)  
        self.label_encoder = Encoder(label_dim)  
      
    def forward(self, features, labels):  
        z_x = self.feature_encoder(features)     
        z_y = self.label_encoder(labels)        
        z_x = F.normalize(z_x, p=2, dim=1)  
        z_y = F.normalize(z_y, p=2, dim=1)  
        similarity_matrix = torch.matmul(z_x, z_y.T)   
        return similarity_matrix  
  
# 获取特征和标签的维度  
feature_dim = x_train.shape[1]  
label_dim = y_train_vectors.shape[1]  
  
# 模型初始化

# 分批次训练函数  
def train_model(model, train_loader, criterion, optimizer, device, epochs=50):  
    model.train()  
    for epoch in range(epochs):  
        total_loss = 0.0  
        batch_count = 0  
        for batch_idx, (features, label_vectors, binary_labels) in enumerate(train_loader):  
            features = features.to(device)  
            label_vectors = label_vectors.to(device)  
            binary_labels = binary_labels.to(device)  
            optimizer.zero_grad()  
            similarity_matrix = model(features, label_vectors)  
            loss = criterion(similarity_matrix, binary_labels)  
            loss.backward()  
            optimizer.step()  
            total_loss += loss.item()  
            batch_count += 1  
        avg_loss = total_loss / batch_count  
        print(f'Epoch {epoch+1}/{epochs} completed. Average Loss: {avg_loss:.6f}')  
      
    return model  
  
# 评估函数  
def evaluate_model(model, test_loader, device):  
    model.eval()  
    total_loss = 0.0  
    batch_count = 0  
    with torch.no_grad():  
        for features, label_vectors, binary_labels in test_loader:  
            features = features.to(device)  
            label_vectors = label_vectors.to(device)  
            binary_labels = binary_labels.to(device)  
            similarity_matrix = model(features, label_vectors)  
            loss = criterion(similarity_matrix, binary_labels)  
            total_loss += loss.item()  
            batch_count += 1  
    avg_loss = total_loss / batch_count  
    print(f'Test Loss: {avg_loss:.6f}')  
    return avg_loss  
  
# 开始训练

# 评估模型

test_loss = evaluate_model(trained_model, test_loader, device)

**核心步骤** ：
- 统计每种攻击的平均特征向量
- 双塔映射同时处理标签和真实流量并压缩到32维 拉到一个公共的空间
- 利用BACLoss把正常流量拉进正常平均原型 推开远离异常平均原型
- 最终测试 看流量与哪一类的平均特征向量余弦相似度最大 就归为哪一类


## 第二步：创新点分析

### 创新点一：标签的语义化增强
- **传统做法：**
	- 标签是正交的 没有物理意义 模型不知道DOS和Probe有什么本质区别
- **本文做法：**
	- 计算了每一类攻击在训练集上的统计均值
	- label encoder接受的不是0和1 而是**这一类攻击的平均特征向量** 

### 创新点二：双向锚点对比损失 （BACLoss）
- 传统做法：无监督 通过数据增强制造正样本对
- 本文做法：
	- **监督对比**
	- **双向约束：** 样本找原型 原型找样本
	- **不对称关注：** 拼命把正常流量压缩的很紧 建立严格的**白名单** 

### 创新点三：从分类到度量的范式转移

- **传统的NIDS**：`Feature -> Model -> Probability`
- **本文的NIDS**：`Feature -> Encoder -> Feature Space <- Encoder <- Prototype` **泛化性强** 


## 2.5

**分析效率低的原因：** 
在特征筛选这一步 做法过于冗余
为了解决特征冗余问题 必须**由简入繁地迭代** 所以每次评估一个特征 都需要**重新训练一遍神经网络** 时间复杂度：$O(N*TrainTime)$ 

**过于保守的原因：** 算法最后留下来太多的特征 剔除不够狠 是因为**Encoder的拟合能力太强了** 导致即使特征很烂 Encoder也可以让loss降下去 导致$derta Loss$ 很小
算法认为该特征不重要 或者相反 因为随机噪声也没能干扰到模型 所以阈值太低 什么特征都舍不得抛弃 

### 优化算法的可能性？

**1. 之前的办法每删除一个特征就重新训练模型** 那不如不要重练
- 输入层后面加一个**权重层** 每一个特征对应一个权重 
- 在loss里面加一个正则项 强迫大部分$w_i$ 变成0
- **结果：** 只需要训练一次 看谁的权重是0 谁就是没用的特征（思路借鉴来源：FSNID）
- 效果：复杂度由$O(N)$ 变成$O(1)$ 

**2:针对BACLoss的优化**
- **如果正常样本对都是一样的** 导致梯度消失 模型最后躺平
- 那么算loss的时候就可以不要平均算 专门挑**最难分** 的样本算loss 加大权重 强迫模型不能偷懒 必须分清楚模棱两可的边界样本



## 第三步：个人意见（？）

### 逻辑漏洞？
- 预处理阶段计算了每一类攻击的均值向量 但是**网络攻击流量模态多样化** 用一个均值代表一类复杂的攻击 不会导致**欠拟合**吗

### 数据泄漏/静态缺陷？
- 测试阶段依旧使用的是训练集计算出来的 `class_mean_vectors` 但是如果测试集出现了严重的概念漂移 模型就会变得没有适应性


## 第四点：个人思考改进可能性

#### 可学习的原型：
- 之前的均值是算死的 不可动
- 改进思考：把y_vector的生成过程去掉 在网络里随机初始化一个矩阵 让其参与**梯度下降**  这样模型就可以在训练过程中摸索出来最完美的5个中心点 （从FSNID借鉴来的想法）

#### 引入对抗训练：
- 之前的模型只学习了简单的样本
- 改进思考：训练时给流量特征x增加轻微扰动 强迫模型面对对抗样本也能匹配到正确的原型 **可以增强模型的鲁棒性**
